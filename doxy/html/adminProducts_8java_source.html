61 void loadDrinks() {
62 try {
63 List<Drink> drinks = apiClient.getAllDrinks();
64 for (Drink drink : drinks) {
65 modelDrink.addRow(new Object[]{
66 drink.getDrinksId(),
67 drink.getItemDrinks(),
68 drink.getDrinksPrice()
69 });
70 }
71 jTable1.setModel(modelDrink);
72 } catch (Exception ex) {
73 JOptionPane.showMessageDialog(null, "Error loading drinks: " + ex.getMessage());
74 }
75 }

80 void loadAppetizer() {
